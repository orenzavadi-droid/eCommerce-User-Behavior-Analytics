WITH funnel_stages AS (
    SELECT 
        COUNT(DISTINCT CASE WHEN event_type = 'view' THEN user_id END) AS total_views,
        COUNT(DISTINCT CASE WHEN event_type = 'cart' THEN user_id END) AS total_carts,
        COUNT(DISTINCT CASE WHEN event_type = 'purchase' THEN user_id END) AS total_purchases
    FROM 
        ecommerce_events
)
SELECT 
    total_views,
    total_carts,
    total_purchases,
    ROUND((total_carts::NUMERIC / total_views) * 100, 2) AS view_to_cart_rate,
    ROUND((total_purchases::NUMERIC / total_carts) * 100, 2) AS cart_to_purchase_rate,
    ROUND((total_purchases::NUMERIC / total_views) * 100, 2) AS overall_conversion_rate
FROM 
    funnel_stages;
